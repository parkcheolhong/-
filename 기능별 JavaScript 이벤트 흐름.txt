## 🎴 표현 카드

```js
document.querySelectorAll('.emotion-card').forEach(card => {
  card.addEventListener('click', () => {
    const emotion = card.dataset.emotion;
    showExpressionExamples(emotion);
  });
});

function showExpressionExamples(emotion) {
  // 예시 출력 + 피드백 버튼 활성화
  document.getElementById('expression-output').innerText = `${emotion} 표현 예시를 보여줍니다.`;
}
```

## 🧠 챌린지

```js
let xp = 0;

document.getElementById('record-btn').addEventListener('click', () => {
  startRecording();
});

function startRecording() {
  // 음성 녹음 시작
  // 녹음 완료 후 XP 증가
  xp += 20;
  document.getElementById('xp-bar').value = xp;
  document.getElementById('xp-label').innerText = `XP: ${xp}`;
}
```

## 🌍 지도

```js
document.getElementById('add-expression').addEventListener('submit', (e) => {
  e.preventDefault();
  const country = e.target.country.value;
  const expression = e.target.querySelector('input').value;
  addPinToMap(country, expression);
});

function addPinToMap(country, expression) {
  // 지도에 핀 추가 + 표현 저장
  console.log(`국가: ${country}, 표현: ${expression}`);
}
```

## 🎥 인터뷰

```js
document.getElementById('generate-intro').addEventListener('click', () => {
  const answer = document.querySelector('textarea').value;
  generateGPTIntro(answer);
});

function generateGPTIntro(text) {
  // GPT API 호출 → 소개글 생성
  document.getElementById('gpt-output').innerText = `GPT 소개글: ${text}에 대한 감정 표현입니다.`;
}
```

## 📰 피드

```js
document.querySelectorAll('.feed-card button').forEach(btn => {
  btn.addEventListener('click', () => {
    const action = btn.innerText;
    handleFeedAction(action);
  });
});

function handleFeedAction(action) {
  // 좋아요 또는 댓글 처리
  console.log(`피드 반응: ${action}`);
}
```

## 🎯 기술 팁

- 이벤트 바인딩: `addEventListener`  
- 상태 관리: `let`, `const`, `DOM 업데이트`  
- API 연동: `fetch`, `async/await`  
- 음성 인식: `Web Speech API`, `Whisper API`  
- 지도 인터랙션: `Leaflet.js`, `D3.js`
